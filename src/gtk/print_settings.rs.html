use glib::translate::*;
use gtk_sys;

use PageRange;
use PrintSettings;

impl PrintSettings {
    /// Sets the value of `GTK_PRINT_SETTINGS_PAGE_RANGES`.
    /// ## `page_ranges`
    /// an array of ``GtkPageRanges``
    /// ## `num_ranges`
    /// the length of `page_ranges`
    pub fn set_page_ranges(&self, page_ranges: &[PageRange]) {
        let num_ranges = page_ranges.len() as i32;
        unsafe {
            gtk_sys::gtk_print_settings_set_page_ranges(
                self.to_glib_none().0,
                mut_override(page_ranges.as_ptr() as *const _),
                num_ranges,
            );
        }
    }
}
